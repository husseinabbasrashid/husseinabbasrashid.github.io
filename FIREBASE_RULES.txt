/* توصية لقواعد Firestore (MUST REVIEW AND TUNE BEFORE PRODUCTION) */

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // رسائل عامة: القراءة عامة، الإنشاء يتطلب مستخدم موثوق (auth) - لحماية من السبام
    match /public_messages/{messageId} {
      allow read: if true;
      // السماح بالكتابة للمستخدمين المصادقين فقط مع تحقق حد الطول
      allow create: if request.auth != null
                    && request.resource.data.keys().hasOnly(['name','text','createdAt','pinned'])
                    && request.resource.data.text is string
                    && request.resource.data.text.size() <= 1000;
      // التعديل/الحذف يقتصر على المسؤولين (اعطاء خاصية admin عبر Custom Claims)
      allow update, delete: if request.auth != null && request.auth.token.admin == true;
    }

    // تقارير الإبلاغ: إنشاء مسموح للمستخدمين المصادقين، القراءة والحذف للمشرفين
    match /reports/{reportId} {
      allow create: if request.auth != null;
      allow read, delete: if request.auth != null && request.auth.token.admin == true;
    }

    // قاعدة افتراضية صارمة
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

/*
ملاحظات أمان:
- هذه القواعد تفترض استخدام Firebase Authentication.
- لتطبيق rate-limiting على مستوى الخادم: استخدم Cloud Functions لرصد الكتابات ومنع مستخدمين يرسلون بسرعة كبيرة (مثال في next file).
- امنح المشرفين Custom Claim 'admin':
  admin.auth().setCustomUserClaims(uid, {admin: true});
*/
